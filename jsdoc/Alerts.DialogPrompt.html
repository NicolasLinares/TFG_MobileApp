Este componente permite introducir el nuevo nombre para una nota de voz

Example
import { DialogPrompt } from '_alerts';

 ...

 render() {
     return (
         <DialogPrompt 
             value={myValue}
             onChangeText={value => doSomething(value)}
             visible={true}
             showError={false}
             onCancel={() => doSomething()}
             onAccept={() => doSomething()}
         />  
     )
 }
